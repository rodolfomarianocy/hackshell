#HackShell - Generator of some different reverse shells:

Reverse Shell:
    a - Bash  
    b - NetCat  
    c - PHP  
    d - Python  
    e - Perl  
    f - Ruby
    g - nodejs

Type:
    d = default
    b64 = base64
    urle = urlencode
    int = integer
    octa = octadeximal
    hex = hexadecimal
    jlre = java.lang.Runtime.exec
    func = function (only nodejs)

---
### ex: hackshell 1a ip port long
